Manual Console Commands

Copy and paste these into your browser console on the admin profile page:

Find All Images

const images = document.querySelectorAll('img'); console.log(`Found ${images.length} images`); images.forEach((img, i) => { console.log(`${i+1}. ${img.src}`); });

Find Broken Images

const broken = Array.from(document.querySelectorAll('img')).filter(img => { return img.complete && img.naturalWidth === 0; }); console.log(`Found ${broken.length} broken images`); broken.forEach((img, i) => { console.error(`BROKEN ${i+1}: ${img.src}`); });
